ENCRYPTION PROGRAMME

this is my code for a simple symmetrical encryption programme.

import random
MODE = input("mode    ")
TEXT = input("text    ")
PASSWORD = input("password    ")


if MODE == "e":

    random.seed(PASSWORD)
    output = ""
    for letter in TEXT:
        output += chr(ord(letter) + random.randint(1000, 10000))
    TEXT = output

else:

    random.seed(PASSWORD)
    output = ""
    for letter in TEXT:
        output += chr(ord(letter) - random.randint(1000, 10000))
    TEXT = output


print(TEXT)
