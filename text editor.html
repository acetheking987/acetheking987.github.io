TEXT EDITOR

this is my code for a simple text editor, some screenshots of the text editor are shown below the code.

import tkinter
from tkinter import font
from tkinter import ttk
import easygui
tk = tkinter.Tk()
tk.geometry("650x650")
FONT = ("Comic Sans MS", 10)
FONTS = list(font.families())
FONT_LIST = []
FONTS.sort()
SIZE = []
CHARACTERS = ["math symbols go here"]


buttons = tkinter.Frame(tk)
font_holder = tkinter.Frame(tk)
main_box = tkinter.Text(tk, font = FONT)
scale_var = tkinter.StringVar()
font_var = tkinter.StringVar()
comboboxsize = ttk.Combobox(font_holder, textvar = font_var, state = "readonly")
combobox = ttk.Combobox(font_holder, textvar = scale_var, state = "readonly")
comboboxcharacters = ttk.Combobox(font_holder, state = "readonly")
version = tkinter.Label(buttons, text = "version: 1.2")


for item in FONTS:
    FONT_LIST.append(item)

for i in range(100):
    SIZE.append(i + 1)

def save():
    dir = easygui.filesavebox("save", "save", "*.txt")
    open(f"{dir}.txt", "w").write(main_box.get("1.0",'end-1c'))

def open_file():
    dir = easygui.fileopenbox("open", "open", "*.txt")
    file = open(f"{dir}", "r").read()
    main_box.delete("1.0", "end")
    main_box.insert(tkinter.END, file)

def update_font(index, value, op):
    global FONT
    try:
        FONT = (combobox.get(), int(comboboxsize.get()))
    except:
        pass
    main_box.config(font = FONT)

def add():
    main_box.insert(tkinter.END, comboboxcharacters.get())


scale_var.trace('w', update_font)
font_var.trace('w', update_font)
combobox['values'] = FONT_LIST
combobox.current(84)
comboboxsize['values'] = SIZE
comboboxsize.current(9)
comboboxcharacters['values'] = CHARACTERS
comboboxcharacters.current(0)
insert_button = tkinter.Button(font_holder, text = "insert", command = add)
save_button = tkinter.Button(buttons, text = "save", command = save)
open_button = tkinter.Button(buttons, text = "open", command = open_file)


buttons.pack(anchor = tkinter.NW)
font_holder.pack(anchor = tkinter.NW)
comboboxsize.pack(side = tkinter.LEFT, padx = 2)
combobox.pack(side = tkinter.LEFT, padx = 0)
comboboxcharacters.pack(side = tkinter.LEFT, padx = 2)
insert_button.pack(side = tkinter.LEFT, padx = 0)
save_button.pack(side = tkinter.LEFT, padx = 2)
open_button.pack(side = tkinter.LEFT, padx = 0)
version.pack(side = tkinter.LEFT, padx = 2)
main_box.pack(fill = "both", expand = True)


tk.mainloop()
